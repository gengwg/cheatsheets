To use a static IP VM from DCO. (dhcp seems no issue.)

vi /etc/sysconfig/network
change gateway

NETWORKING=yes
HOSTNAME=ganglia2
#GATEWAY=10.30.16.1
GATEWAY=192.168.0.1
                     
vi /etc/sysconfig/network-scripts/ifcfg-eth0
change ip

DEVICE=eth0
TYPE=Ethernet
ONBOOT=yes
NM_CONTROLLED=yes
#BOOTPROTO=dhcp
BOOTPROTO=static
#IPADDR=10.30.24.73
IPADDR=192.168.0.45

vi /etc/resolv.conf
change nameserver

search Home
nameserver 192.168.0.1
nameserver 205.171.2.25

then,
service network restart.
